C) Create a Flipkart discount calculator that asks the user for the total cart amount. Use nested if statements to check: if amount > 2000, apply 20% discount; else if amount > 1000, apply 10% discount; else, no discount. Print the final amount to pay.

#include<stdio.h>
#include<conio.h>
void main()

{
	float amount;
	float discount;
	float Finalvalue;
		printf( "Enter your Total cart amount" );
		scanf("%f" , &amount);
		
    if (amount > 1000)
    {
        if (amount > 2000)
        {
            discount = amount * 20 / 100;
        }
        else
        {
            discount = amount * 10 / 100;
        }
    }
    else
    {
        discount = 0;
    }

    Finalvalue = amount - discount;
