B) Create a constant variable to store the GST rate (for example, 18%) and use it to calculate the final price of a Zomato order with a given base price.

#include<stdio.h>
#include<conio.h>
void main()

{     
    const float GST = 18;
    float baseprice = 300;
    float gstamount = (baseprice * GST) / 100;
    float finalprice = baseprice + gstamount;
    
    printf("\n GST is : %f" , GST);
    printf("\n Base price is : %f" , baseprice);
    printf("\n GST amount is : %f" , gstamount);
    printf("\n Final Price is : %f" , finalprice);
